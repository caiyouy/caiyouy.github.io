{
    int AREA=recArea(A,B,C,D)+recArea(E,F,G,H);
    if(A<=E)
    {
        if(C<=E)
            return AREA;
        if(B>=F)
        {
            if(B>=H)
                return AREA;

}

                return;

                                bit[i]=1;

                                bit[i]=0;

        }

    return *p==*s&&isMatch(p+1, s+1);
