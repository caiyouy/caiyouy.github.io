#include <stdio.h>

int main()
{
    char Line[1000];

    while((c=getchar())!=EOF)
    {
        Line[count++]=c;
        if(c=='\n')  //读入一行
        {
            //进行判断
            if(count<=10) //输出
            {
                for(int i=0;i<count;i++)
                {
                    putchar(Line[i]);
                }
            }
            else //跳过
            {
            }
            count=0;
        }
    }
    return 0;
}

牛顿迭代及二分法

代码点评一:

#include <stdio.h>
#include <math.h>
#define e0 1e-13
double f(double x);
double f_(double x);

void Newton(double x0){
    static int step=0;                 // static 的作用?
    if(fabs(x0*exp(x0)-1) < e0)
        printf("Newton method, root: %f ,iterations: %d\n",x0,step);
    else{
        x0=x0-f(x0)/f_(x0);
        step++;
        Newton(x0);
    }
}

void Bisec(double a,double b){
    static int step=0;
    double c;
    if(fabs(b*exp(b)-1) < e0)
        printf("Bisection method, root: %f ,iterations: %d",b,step);
    else {
        c=(a+b)*0.5;
        step++;
        if(f(c)*f(a)<0)
            Bisec(a,c);
        else
            Bisec(c,b);
    }
}

double f(double x0){
    return x0*exp(x0)-1;
}
double f_(double x){
    return (x+1.)*exp(x);
}
int main(){
    Newton(0.5);
    Bisec(0.5,0.6);
    return 0;
}

int valid(int year, int month, int day)
{
    // test year
    if(year<=0) return 0; //Not exactly accurate.
    // test month
    if(month<=0||month>=13) return 0;
    // test day
    if(day<=0) return 0;
    if(month== 1&&day>31) return 0;
